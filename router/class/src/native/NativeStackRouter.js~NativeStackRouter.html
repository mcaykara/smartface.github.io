    src/native/NativeRouterBase.js~Router → NativeStackRouter

    Creates NavigationController and manages its behavours and routes.

    Example:

    const {NativeStackRouter, Route} = require('@smartface/router');
const Image = require('sf-core/ui/image');
const Color = require('sf-core/ui/color');

var router = Router.of({
 path: "/",

 routes: [
   NativeStackRouter.of(
     path: '/pages',
     headerBarParams: () => ({
       ios: {
         translucent: true,
         alpha: 1
       },
       backgroundColor: Color.BLUE,
       visible: true
     }),
     routes: [
       Route.of({
         path: "/pages/page1",
         build((router, route) => {
           const Page1 = require('/pages/Page1');
           return new Page1(state.data, router);
         })
       }),
       Route.of({
         path: "/pages/page2",
         build((router, route) => {
           const Page2 = require('/pages/Page2');
           return new Page2(state.data, router);
         });
       });
     ]
   )]
});
